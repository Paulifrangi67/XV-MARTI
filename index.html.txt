    <script>
        document.getElementById('uploadBtn').addEventListener('click', async () => {
            const fileInput = document.createElement('input');
            fileInput.type = 'file';
            fileInput.accept = 'image/*';
            fileInput.onchange = async (event) => {
                const file = event.target.files[0];
                if (!file) return;

                const formData = new FormData();
                formData.append('file', file);

                try {
                    const response = await fetch('https://api.uploadcare.com/base/', { // ðŸ”„ Reemplazar por FileStack si querÃ©s
                        method: 'POST',
                        body: formData,
                        headers: { 'Authorization': 'public-key cbad6a9d84d52ee909cb' }
                    });
                    const data = await response.json();
                    const imgUrl = data.file_url || data.url;
                    const imgElement = document.createElement('img');
                    imgElement.src = imgUrl;
                    document.getElementById('uploadedImages').appendChild(imgElement);
                } catch (error) {
                    alert('Error subiendo la imagen');
                }
            };
            fileInput.click();
        });
    </script>
